id: api_to_s3_parquet
namespace: company.team

tasks:
  # 1. Download data from the API
  - id: download_json
    type: io.kestra.plugin.core.http.Download
    uri: "https://api.example.com/data"

  # 2. Convert the downloaded JSON to Kestra's internal Ion format
  - id: json_to_ion
    type: io.kestra.plugin.serdes.json.JsonToIon
    from: "{{ outputs.download_json.uri }}"
    newLine: false # Set to true if your API returns NDJSON (newline delimited JSON)

  # 3. Convert Ion to Parquet
  - id: ion_to_parquet
    type: io.kestra.plugin.serdes.parquet.IonToParquet
    from: "{{ outputs.json_to_ion.uri }}"
    # Defining a schema is often required for Parquet to know column types
    schema: |
      {
        "type": "record",
        "name": "ApiData",
        "namespace": "com.example",
        "fields": [
          {"name": "id", "type": "int"},
          {"name": "name", "type": "string"},
          {"name": "timestamp", "type": "long"}
        ]
      }

  # 4. Upload the result to S3
  - id: upload_to_s3
    type: io.kestra.plugin.aws.s3.Upload
    from: "{{ outputs.ion_to_parquet.uri }}"
    bucket: "my-data-bucket"
    key: "raw/data/{{ now() | date('yyyy/MM/dd') }}/output.parquet"
    accessKeyId: "{{ secret('AWS_ACCESS_KEY_ID') }}"
    secretKeyId: "{{ secret('AWS_SECRET_ACCESS_KEY') }}"
    region: "us-east-1"
